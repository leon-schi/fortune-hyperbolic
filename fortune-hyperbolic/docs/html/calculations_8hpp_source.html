91 template <typename _float_T>
92 struct cosine {
93 _float_T amp, phase, c; // amplitude, phase, and constant c
94 cosine(_float_T amp, _float_T phase, _float_T c) : amp(amp), phase(phase), c(c) {}
95
96 cosine<_float_T> operator+(const cosine<_float_T>& a) const {
97 // the phase returned is always in [0, 2pi]
98 _float_T x = cos(a.phase)*a.amp;
99 x += cos(phase)*amp;
100 _float_T y = sin(a.phase)*a.amp;
101 y += sin(phase)*amp;
102
103 _float_T result_amp = sqrt(pow(x, 2) + pow(y,2));
104 _float_T result_phase = atan2(y, x);
105 if (result_phase < 0) result_phase += 2*M_PI;
106 return {result_amp, result_phase, a.c + c};
107 }
108
109 cosine<_float_T> operator-(const cosine<_float_T>& a) const{
110 return *this + cosine(-a.amp, a.phase, -a.c);
111 }
112
113 cosine<_float_T> operator *(const _float_T a) const {
114 return {a * amp, phase, a * c};
115 }
116
117 _float_T operator ()(_float_T x) const {
118 return amp*cos(x + phase) + c;
119 }

121 [[nodiscard]] std::pair<_float_T, _float_T> zeros() const {
122 // returns zeros of the cosine function a in the interval [0, 2 pi)
123 _float_T z = acos(-c/amp);
124 _float_T z1 = clip<_float_T>(z - phase);
125 _float_T z2 = clip<_float_T>(2*M_PI - z - phase);

146 void calc_definition() {
147 // returns two angular coordinates between which the bisector is defined
148 _float_T phi = acos(numerator/denominator.amp); // phi is in [0, pi/2]
149 theta_start = clip<_float_T>(2*M_PI - phi - denominator.phase);
150 theta_end = clip<_float_T>(phi - denominator.phase);
151 }

153 // constructs the object from the two points that define it
154 Bisector(const Point<_float_T>* s, const Point<_float_T>* t) {
155 if (s->r > t->r)
156 std::swap(s, t);
157 numerator = cosh(t->r) - cosh(s->r);
158
159 if (numerator == 0.0) {
160 is_straight = true;
161 straight_angle = (s->theta + t->theta) / 2.0;
162 } else {
163 is_straight = false;
164 denominator =
165 cosine<_float_T>(sinh(t->r), -t->theta, 0) +
166 cosine<_float_T>(-sinh(s->r), -s->theta, 0);
167 calc_definition();
168 }
169 }
