70 void render_edge(Point<double>* from, Point<double>* to, HyperboloidBisector<double>& b, list<CartesianPoint>& p, bool ccw=true) const {
71 double dt = options.resolution * scale;
72 Point u_native(b.u);
73 double t = (from) ? distance<double>(*from, u_native) : 0;
74 double t_end = (to) ? distance<double>(*to, u_native) : DBL_MAX;
75
76 HyperboloidVec<double> v = b.v;
77 Point v_polar(v);
78 Point u_polar(b.u);
79 double theta = clip(v_polar.theta - u_polar.theta);
80 if ((ccw && theta >= M_PI) || (!ccw && theta <= M_PI))
81 v = v*(-1);
82
83 while (t < t_end) {
84 Point point(b.u*cosh(t) + v*sinh(t));
85 p.emplace_back(point);
86 t += dt;
87 if (out_of_bounds(point)) break;
88 //if (point.r*scale*1.414 > width) break;
89 }
90
91 if (from)
92 p.emplace_front(*from);
93 if (to)
94 p.emplace_back(*to);
95 }

97 void add_edge(Edge& e, Path& p) const {
98 HyperboloidBisector<double> b(e.siteA.point, e.siteB.point);
99 Point q(b.u);
100 if (e.edgeType == EdgeType::BIDIRECTIONAL) {
101 list<CartesianPoint> cw, ccw;
102 render_edge(nullptr, e.firstVertex, b, ccw, true);
103 render_edge(nullptr, e.secondVertex, b, cw, false);
104 cw.emplace_front(q);
105 cw.reverse();
106 p.insert(p.begin(), ccw.begin(), ccw.end());
107 p.insert(p.begin(), cw.begin(), cw.end());
108 }
109 else {
110 list<CartesianPoint> p_list;
111 render_edge(e.firstVertex, e.secondVertex, b, p_list, e.edgeType == EdgeType::CCW);
112 p.insert(p.begin(), p_list.begin(), p_list.end());
113 }
